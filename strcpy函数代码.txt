#include<stdlib.h>
char* StrCpy(char* src) {
	int length=strlen(src);
	char *p = (char *)malloc(sizeof(char) * (length+1));
	for (int i = 0; i < length; i++) {
		p[i] = src[i];
	}
	p[length] = '\0';
	return p;
}
